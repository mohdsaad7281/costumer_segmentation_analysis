-- Step 1: Create clean table
CREATE TABLE retail_clean AS
SELECT 
    Invoice, StockCode, Description,
    CAST(Quantity AS SIGNED) AS Quantity,
    STR_TO_DATE(InvoiceDate, '%d-%m-%Y %H:%i') AS InvoiceDate,
    CAST(Price AS DECIMAL(10,2)) AS Price,
    CustomerID, Country
FROM online_retail
WHERE CustomerID IS NOT NULL
    AND CustomerID != ''
    AND CAST(Quantity AS SIGNED) > 0
    AND CAST(Price AS DECIMAL(10,2)) > 0;

-- Step 2: RFM Base
CREATE TABLE rfm_base AS
SELECT
    CustomerID, Country,
    DATEDIFF('2011-12-05', MAX(InvoiceDate)) AS Recency,
    COUNT(DISTINCT Invoice) AS Frequency,
    ROUND(SUM(Quantity * Price), 2) AS Monetary
FROM retail_clean
GROUP BY CustomerID, Country;

-- Step 3: RFM Scores
CREATE TABLE rfm_scores AS
SELECT
    CustomerID, Country, Recency, Frequency, Monetary,
    NTILE(4) OVER (ORDER BY Recency DESC) AS R_Score,
    NTILE(4) OVER (ORDER BY Frequency ASC) AS F_Score,
    NTILE(4) OVER (ORDER BY Monetary ASC) AS M_Score
FROM rfm_base;

-- Step 4: Segments
CREATE TABLE rfm_segments AS
SELECT
    CustomerID, Country, Recency, Frequency, Monetary,
    R_Score, F_Score, M_Score,
    (R_Score + F_Score + M_Score) AS RFM_Total,
    CASE
        WHEN (R_Score + F_Score + M_Score) >= 10 THEN 'Champions'
        WHEN (R_Score + F_Score + M_Score) >= 8 THEN 'Loyal Customers'
        WHEN (R_Score + F_Score + M_Score) >= 6 THEN 'Potential Loyalists'
        WHEN (R_Score + F_Score + M_Score) >= 5 THEN 'At Risk'
        WHEN (R_Score + F_Score + M_Score) >= 4 THEN 'Hibernating'
        ELSE 'Lost'
    END AS Segment
FROM rfm_scores;

-- Step 5: Cohort Query
SELECT 
    c.CustomerID,
    c.cohort_month,
    DATE_FORMAT(r.InvoiceDate, '%Y-%m') AS invoice_month,
    COUNT(DISTINCT r.Invoice) AS orders
FROM retail_clean r
JOIN (
    SELECT CustomerID,
        DATE_FORMAT(MIN(InvoiceDate), '%Y-%m') AS cohort_month
    FROM retail_clean
    GROUP BY CustomerID
) c ON r.CustomerID = c.CustomerID
GROUP BY c.CustomerID, c.cohort_month, DATE_FORMAT(r.InvoiceDate, '%Y-%m');
